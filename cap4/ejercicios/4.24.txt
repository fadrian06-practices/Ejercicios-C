	a) printf( “%d”, j==1 );
		1
		
	b) printf( “%d”, j==3 );
		0
		
	c) printf( “%d”, i >= 1 && j < 4 );
		1
	
	d) printf( “%d”, m >= 99 && k < m );
		0
	
	e) printf( “%d”, j >= i || k == m );
		1
	
	f) printf( “%d”, k + m < j || 3 – j >= k );
		0
	
	g) printf( “%d”, !m );
		0
	
	h) printf( “%d”, !( j – m ) );
		1
	
	i) printf( “%d”, !( k > m ) );
		0
	
	j) printf( “%d”, !( j > k ) );
		1
